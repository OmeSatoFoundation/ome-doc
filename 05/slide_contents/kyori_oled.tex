\begin{frame}[fragile]
    \frametitle{距離センサーのプログラム}
    \begin{lstlisting}[title=\textasciitilde/05/kyori.hsp]
    #include "hsp3dish.as"
    #include "rpz-gpio.as"

    spiopen 0

    *main

	    data = spiget(0,0)
	    <#red#kyori = 3952640 / (875*data + 22272)#>
	    res = "距離 : "+kyori
	
	    redraw 0
	    font "",20
	    pos 30,30
	    mes res
	    redraw 1

	    wait 10
	    goto *main

    spiclose 0
    \end{lstlisting}
    \textpageref{18}
\end{frame}
