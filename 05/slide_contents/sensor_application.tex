\begin{frame}[fragile]
    \frametitle{ボリュームを使って画像を動かそう(angle.hsp)}
\begin{lstlisting}
#include "hsp3dish.as"
#include "rpz-gpio.as"

celload("hyou.png"),2
p = 50

spiopen 0

*main
	data = spiget(0,0)
	p = rasp_map(data, 0, 1023, 0, 440)

	redraw 0
	pos 100,100
	mes data
	mes p
	pos 0,0
	celput 2
	redraw 1

	wait 10	
	goto *main

spiclose 0
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
    \frametitle{LEDの明るさを変えよう(pwm.hsp)}
\begin{lstlisting}
#include "hsp3dish.as"
#include "rpz-gpio.as"

redraw 0
redraw 1

repeat
gpio 27,1

repeat 200
if cnt < 10 : gpio 17,1 : else : gpio 17,0
if cnt < 50 : gpio 18,1 : else : gpio 18,0
if cnt < 100 : gpio 22,1 : else : gpio 22,0
loop

await 1
loop

gpio 17,0
gpio 18,0
gpio 22,0
gpio 27,0
stop
\end{lstlisting}
\end{frame}
