\begin{frame}[fragile]
    \frametitle{デジタル入力センサを使った\\プログラム}
    \begin{lstlisting}[title=\textasciitilde/05/digin.hsp]
    #include "hsp3dish.as"
    #include "rpz-gpio.as"

    *main
	    redraw 0
	    pos 30,30
	    mes "デジタル入力そうちでLEDを光らせよう"
	    redraw 1
	    if gpioin(24)=1 : goto *digital_on
	    goto *digital_off
        
    *digital_on
	    gpio 23, 1
	    wait 10
	    goto *main

    *digital_off
	    gpio 23, 0
	    wait 10
	    goto *main 
    \end{lstlisting}
    \textpageref{14}
\end{frame}
